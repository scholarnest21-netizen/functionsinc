Structures in C

struct keyword · user-defined data type
What is a Structure?
A structure is a user-defined data type in C that allows grouping of variables of different data types under a single name. Unlike arrays (which store elements of the same type), structures can combine int, float, char, and other types together as a single logical unit called a record.

Structures are declared using the struct keyword. Each variable inside a structure is called a member or field. Once defined, a structure acts as a new data type from which variables can be created.
📌 Structures are fundamental to data modeling — they let you represent real-world entities like a Student, Employee, or Book as a single compound variable.
CStructure Declaration Syntax
struct StructureName {
    datatype member1;
    datatype member2;
    // ... more members
};

/* Example: Student structure */
struct Student {
    int   rollNo;
    char  name[50];
    float marks;
    char  grade;
};
Declaring Structure Variables & Accessing Members
Structure variables are declared after the structure definition. Members are accessed using the dot operator (.).
CExample 1 – Basic Structure
#include <stdio.h>
#include <string.h>

struct Student {
    int   rollNo;
    char  name[50];
    float marks;
};

int main() {
    struct Student s1;          // declare variable

    s1.rollNo = 101;
    strcpy(s1.name, "Alice");
    s1.marks = 92.5;

    printf("Roll No : %d\n", s1.rollNo);
    printf("Name    : %s\n", s1.name);
    printf("Marks   : %.2f\n", s1.marks);

    return 0;
}
Output
Roll No : 101
Name : Alice
Marks : 92.50
CExample 2 – Initialization at Declaration
#include <stdio.h>

struct Point {
    int x, y;
};

int main() {
    struct Point p1 = {3, 7};    // initializer list
    struct Point p2 = {.x = 10, .y = 20};   // designated

    printf("p1 = (%d, %d)\n", p1.x, p1.y);
    printf("p2 = (%d, %d)\n", p2.x, p2.y);
    return 0;
}
Output
p1 = (3, 7)
p2 = (10, 20)
CExample 3 – Structure as Function Argument (pass by value)
#include <stdio.h>

struct Rectangle {
    float length, width;
};

float area(struct Rectangle r) {
    return r.length * r.width;
}

int main() {
    struct Rectangle rect = {5.0, 3.0};
    printf("Area = %.2f\n", area(rect));
    return 0;
}
Output
Area = 15.00

Nested Structures

struct inside struct · hierarchical data
What are Nested Structures?
A nested structure is a structure that has another structure as one of its members. This allows representing complex, hierarchical real-world data. For example, an Employee structure can contain a Date structure for the date of birth and a Address structure for the address.

Members of nested structures are accessed using the dot operator chained: outer.inner.field.
CExample 1 – Date inside Employee
#include <stdio.h>

struct Date {
    int day, month, year;
};

struct Employee {
    int         id;
    char        name[40];
    struct Date dob;          // nested structure
    float       salary;
};

int main() {
    struct Employee e1 = {1001, "Bob", {15, 8, 1995}, 45000.0};

    printf("ID     : %d\n", e1.id);
    printf("Name   : %s\n", e1.name);
    printf("DOB    : %d/%d/%d\n",
           e1.dob.day, e1.dob.month, e1.dob.year);
    printf("Salary : %.2f\n", e1.salary);
    return 0;
}
Output
ID : 1001
Name : Bob
DOB : 15/8/1995
Salary : 45000.00
CExample 2 – Address inside Student (3-level nesting)
#include <stdio.h>

struct Address {
    char street[50];
    char city[30];
    int  pin;
};

struct Student {
    int            roll;
    char           name[40];
    struct Address addr;
};

int main() {
    struct Student s = {201, "Priya", {"MG Road", "Jaipur", 302001}};
    printf("%s lives at %s, %s - %d\n",
           s.name, s.addr.street, s.addr.city, s.addr.pin);
    return 0;
}
Output
Priya lives at MG Road, Jaipur - 302001

Pointer and Structures

struct pointer · arrow operator → · pass by reference
Structure Pointers
A pointer to a structure stores the address of a structure variable. Instead of the dot operator, the arrow operator (->) is used to access members through a pointer.

Using pointers with structures is more efficient when passing structures to functions, because only the address (4 or 8 bytes) is copied instead of the entire structure.
📌 ptr->member is equivalent to (*ptr).member. Always prefer the arrow operator for readability.
CExample 1 – Pointer to Structure
#include <stdio.h>

struct Book {
    char  title[50];
    char  author[30];
    float price;
};

int main() {
    struct Book  b1 = {"Let Us C", "Kanetkar", 299.0};
    struct Book *ptr = &b1;    // pointer to structure

    /* Accessing via arrow operator */
    printf("Title  : %s\n", ptr->title);
    printf("Author : %s\n", ptr->author);
    printf("Price  : Rs %.2f\n", ptr->price);

    ptr->price = 350.0;    // modifying via pointer
    printf("New Price: Rs %.2f\n", b1.price);
    return 0;
}
Output
Title : Let Us C
Author : Kanetkar
Price : Rs 299.00
New Price: Rs 350.00
CExample 2 – Pass Structure by Pointer to Function
#include <stdio.h>

struct Circle {
    float radius;
};

void doubleRadius(struct Circle *c) {
    c->radius *= 2;    // modifies original
}

int main() {
    struct Circle c1 = {5.0};
    printf("Before: r = %.1f\n", c1.radius);
    doubleRadius(&c1);
    printf("After : r = %.1f\n", c1.radius);
    return 0;
}
Output
Before: r = 5.0
After : r = 10.0

Array of Structures

collection of records · struct arrays
Arrays of Structures
An array of structures allows storing multiple records of the same structure type. Each element of the array is a complete structure variable. This is like a table — each row is a record, each column is a field.
CExample 1 – Student Records (Array of Structures)
#include <stdio.h>

struct Student {
    int   roll;
    char  name[30];
    float marks;
};

int main() {
    struct Student s[3] = {
        {101, "Alice", 88.5},
        {102, "Bob",   75.0},
        {103, "Carol", 91.2}
    };

    printf("%-5s %-10s %s\n", "Roll", "Name", "Marks");
    printf("------------------------\n");
    for (int i = 0; i < 3; i++)
        printf("%-5d %-10s %.1f\n",
               s[i].roll, s[i].name, s[i].marks);
    return 0;
}
Output
Roll Name Marks
------------------------
101 Alice 88.5
102 Bob 75.0
103 Carol 91.2
CExample 2 – Find Highest Marks in Array
#include <stdio.h>

struct Student {
    char  name[30];
    float marks;
};

int main() {
    struct Student s[4] = {
        {"Raj", 78.0}, {"Meena", 95.5},
        {"Tom",  88.0}, {"Sara",  92.0}
    };
    int   topIdx = 0;
    for (int i = 1; i < 4; i++)
        if (s[i].marks > s[topIdx].marks) topIdx = i;
    printf("Topper: %s (%.1f)\n", s[topIdx].name, s[topIdx].marks);
    return 0;
}
Output
Topper: Meena (95.5)

Example Program: Structures & Pointers

comprehensive example · inventory system
A complete inventory management program combining arrays of structures with pointer-based operations.
CInventory System – Struct + Pointer + Array
#include <stdio.h>
#include <string.h>

#define MAX 5

struct Product {
    int   id;
    char  name[30];
    int   qty;
    float price;
};

void display(struct Product *p, int n) {
    printf("\n%-5s %-15s %-8s %s\n", "ID", "Name", "Qty", "Price");
    printf("-----------------------------------\n");
    for (int i = 0; i < n; i++)
        printf("%-5d %-15s %-8d %.2f\n",
               (p+i)->id, (p+i)->name,
               (p+i)->qty, (p+i)->price);
}

float totalValue(struct Product *p, int n) {
    float total = 0;
    for (int i = 0; i < n; i++)
        total += p[i].qty * p[i].price;
    return total;
}

int main() {
    struct Product inv[MAX] = {
        {1, "Laptop",    10, 55000.0},
        {2, "Mouse",     50,   450.0},
        {3, "Keyboard",  30,  1200.0},
        {4, "Monitor",   15, 12000.0},
        {5, "USB Hub",   40,   800.0}
    };

    display(inv, MAX);
    printf("\nTotal Inventory Value: Rs %.2f\n", totalValue(inv, MAX));
    return 0;
}
Output
ID Name Qty Price
-----------------------------------
1 Laptop 10 55000.00
2 Mouse 50 450.00
3 Keyboard 30 1200.00
4 Monitor 15 12000.00
5 USB Hub 40 800.00

Total Inventory Value: Rs 659500.00

Self-Referential Structures

struct pointer to itself · foundation of linked lists & trees
What is a Self-Referential Structure?
A self-referential structure contains a pointer member that points to the same structure type. This is the fundamental building block for dynamic data structures like linked lists, trees, stacks, and queues.

The pointer doesn't contain another structure (that would be infinite size), it contains only the address of another structure of the same type — so the size remains finite.
⚠️ A structure cannot contain itself (infinite recursion), but it can contain a pointer to itself — pointers have a fixed size regardless of what they point to.
CSelf-Referential Structure Syntax
struct Node {
    int          data;
    struct Node *next;    // pointer to same type
};

/* Tree node — two self-referential pointers */
struct TreeNode {
    int              value;
    struct TreeNode *left, *right;
};
CExample – Manually Chaining 3 Nodes
#include <stdio.h>
#include <stdlib.h>

struct Node {
    int          data;
    struct Node *next;
};

int main() {
    struct Node n1, n2, n3;
    n1.data = 10;  n1.next = &n2;
    n2.data = 20;  n2.next = &n3;
    n3.data = 30;  n3.next = NULL;

    struct Node *ptr = &n1;
    printf("Linked list: ");
    while (ptr != NULL) {
        printf("%d ", ptr->data);
        ptr = ptr->next;
    }
    printf("\n");
    return 0;
}
Output
Linked list: 10 20 30

Dynamic Memory Allocation

malloc · calloc · realloc · free · <stdlib.h>
What is Dynamic Memory Allocation?
In C, memory can be allocated in two ways: statically (at compile time, like arrays) or dynamically (at runtime, using heap memory). Dynamic allocation lets programs request memory while running — useful when the size is not known in advance.

The four key functions from <stdlib.h>:
FunctionPurposeReturns
malloc(size)Allocates size bytes (uninitialized)void* or NULL
calloc(n, size)Allocates n×size bytes (zero-initialized)void* or NULL
realloc(ptr, size)Resizes previously allocated blockvoid* or NULL
free(ptr)Releases allocated memory back to heapvoid
CExample 1 – malloc and free
#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, i;
    printf("Enter count: ");
    scanf("%d", &n);

    int *arr = (int*)malloc(n * sizeof(int));
    if (arr == NULL) {
        printf("Memory allocation failed!\n");
        return 1;
    }
    for (i = 0; i < n; i++) arr[i] = (i + 1) * 10;

    printf("Array: ");
    for (i = 0; i < n; i++) printf("%d ", arr[i]);
    printf("\n");

    free(arr);    // release memory
    return 0;
}
Output (for n=4)
Enter count: 4
Array: 10 20 30 40
CExample 2 – calloc (zero-initialized)
#include <stdio.h>
#include <stdlib.h>

int main() {
    int *arr = (int*)calloc(5, sizeof(int));
    printf("calloc (all zeros): ");
    for (int i = 0; i < 5; i++) printf("%d ", arr[i]);
    printf("\n");

    /* Resize to 8 elements */
    arr = (int*)realloc(arr, 8 * sizeof(int));
    printf("After realloc (8 elements, last 3 uninitialized):\n");
    free(arr);
    return 0;
}
Output
calloc (all zeros): 0 0 0 0 0
CExample 3 – Dynamic Array of Structures
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Student { char name[30]; float gpa; };

int main() {
    int n = 3;
    struct Student *s = (struct Student*)
                        malloc(n * sizeof(struct Student));
    strcpy(s[0].name, "Amit");   s[0].gpa = 8.5;
    strcpy(s[1].name, "Neha");   s[1].gpa = 9.1;
    strcpy(s[2].name, "Kiran");  s[2].gpa = 7.8;

    for (int i = 0; i < n; i++)
        printf("%s: GPA %.1f\n", s[i].name, s[i].gpa);

    free(s);
    return 0;
}
Output
Amit: GPA 8.5
Neha: GPA 9.1
Kiran: GPA 7.8

Singly Linked List

nodes · head pointer · insert · delete · traverse
What is a Singly Linked List?
A singly linked list is a dynamic data structure where each element (called a node) stores data and a pointer to the next node. Unlike arrays, linked lists don't require contiguous memory — nodes can be scattered in heap memory and linked via pointers.

Key properties: Head points to the first node. The last node's next pointer is NULL. Traversal goes in one direction only (forward).
✅ Linked lists allow O(1) insertion/deletion at the front and dynamic sizing — advantages over static arrays.
CComplete Singly Linked List – Insert, Delete, Display
#include <stdio.h>
#include <stdlib.h>

struct Node {
    int          data;
    struct Node *next;
};

/* Create a new node */
struct Node* newNode(int val) {
    struct Node *n = (struct Node*)malloc(sizeof(struct Node));
    n->data = val;
    n->next = NULL;
    return n;
}

/* Insert at beginning */
void insertFront(struct Node **head, int val) {
    struct Node *n = newNode(val);
    n->next = *head;
    *head = n;
}

/* Insert at end */
void insertEnd(struct Node **head, int val) {
    struct Node *n = newNode(val);
    if (*head == NULL) { *head = n; return; }
    struct Node *tmp = *head;
    while (tmp->next) tmp = tmp->next;
    tmp->next = n;
}

/* Delete a node by value */
void deleteNode(struct Node **head, int val) {
    struct Node *curr = *head, *prev = NULL;
    while (curr && curr->data != val)
        { prev = curr; curr = curr->next; }
    if (!curr) { printf("%d not found\n", val); return; }
    if (!prev) *head = curr->next;
    else       prev->next = curr->next;
    free(curr);
}

/* Display list */
void display(struct Node *head) {
    printf("List: ");
    while (head) {
        printf("%d", head->data);
        if (head->next) printf(" -> ");
        head = head->next;
    }
    printf(" -> NULL\n");
}

int main() {
    struct Node *head = NULL;
    insertEnd(&head, 10);
    insertEnd(&head, 20);
    insertEnd(&head, 30);
    insertFront(&head, 5);
    display(head);

    deleteNode(&head, 20);
    display(head);
    return 0;
}
Output
List: 5 -> 10 -> 20 -> 30 -> NULL
List: 5 -> 10 -> 30 -> NULL

typedef in C

type alias · cleaner code · struct shorthand
What is typedef?
typedef creates an alias (alternate name) for an existing data type. It doesn't create a new type — it gives an existing type a new, shorter, or more meaningful name. It is most commonly used with structures and pointers to simplify code.
CExample 1 – typedef with Primitive Types
#include <stdio.h>

typedef unsigned int uint;
typedef float        real;
typedef char         String[50];

int main() {
    uint   age    = 21;
    real   gpa    = 8.75;
    String course = "Computer Science";
    printf("Age: %u, GPA: %.2f, Course: %s\n", age, gpa, course);
    return 0;
}
CExample 2 – typedef with Structures (most common usage)
#include <stdio.h>

typedef struct {
    int   roll;
    char  name[30];
    float gpa;
} Student;    // no "struct" keyword needed now

void print(Student *s) {
    printf("%d | %-10s | %.2f\n", s->roll, s->name, s->gpa);
}

int main() {
    Student s1 = {101, "Anjali", 9.0};
    Student s2 = {102, "Rohan",  8.3};
    print(&s1);
    print(&s2);
    return 0;
}
Output
101 | Anjali | 9.00
102 | Rohan | 8.30
CExample 3 – typedef with Linked List Node
typedef struct Node {
    int         data;
    struct Node *next;
} Node;

File Processing in C

fopen · fclose · fprintf · fscanf · fread · fwrite
What are Files?
A file is a collection of related data stored permanently on a secondary storage device (disk). Unlike variables (which lose values when a program ends), files persist data between program runs.

C provides the FILE pointer type (from <stdio.h>) for working with files. Key functions:
FunctionPurpose
fopen(name, mode)Open a file; returns FILE pointer
fclose(fp)Close an open file
fprintf(fp, ...)Write formatted text to file
fscanf(fp, ...)Read formatted text from file
fputc / fgetcWrite/read single character
fputs / fgetsWrite/read string
fwrite / freadBinary block write/read
fseek / ftell / rewindRandom access positioning
feof(fp)Check end-of-file
File Opening Modes
ModeMeaning
"r"Open for reading (file must exist)
"w"Open for writing (creates or overwrites)
"a"Open for appending (adds to end)
"r+"Open for reading and writing
"w+"Open for read/write (overwrites)
"rb"/"wb"Binary read/write modes
Types of File Processing
There are two primary types: Sequential Access (read/write records one after another, from beginning) and Random Access (jump directly to any record using its position). Text files are typically sequential; binary files support efficient random access.
📄

Sequential Access File

text files · read in order · fprintf · fscanf · feof
Sequential Access
In sequential access, data is read or written in order from the beginning of the file to the end — like reading a book page by page. You cannot jump directly to record #50 without reading records 1–49 first.

Sequential access files are typically text files and use fprintf / fscanf for writing and reading formatted data.
CExample 1 – Write to Sequential File
#include <stdio.h>

int main() {
    FILE *fp = fopen("students.txt", "w");
    if (fp == NULL) {
        printf("Cannot open file!\n");
        return 1;
    }
    fprintf(fp, "%d %s %.2f\n", 101, "Alice", 88.5);
    fprintf(fp, "%d %s %.2f\n", 102, "Bob",   75.0);
    fprintf(fp, "%d %s %.2f\n", 103, "Carol", 91.2);
    fclose(fp);
    printf("File written successfully.\n");
    return 0;
}
students.txt contents
101 Alice 88.50
102 Bob 75.00
103 Carol 91.20
CExample 2 – Read from Sequential File
#include <stdio.h>

int main() {
    FILE *fp = fopen("students.txt", "r");
    int  roll; char name[30]; float marks;

    printf("Roll  Name      Marks\n");
    printf("---------------------\n");
    while (fscanf(fp, "%d %s %f", &roll, name, &marks) == 3)
        printf("%-5d %-10s %.2f\n", roll, name, marks);
    fclose(fp);
    return 0;
}
Output
Roll Name Marks
---------------------
101 Alice 88.50
102 Bob 75.00
103 Carol 91.20
CExample 3 – Character-by-Character Copy
#include <stdio.h>

int main() {
    FILE *src  = fopen("source.txt",  "r");
    FILE *dest = fopen("dest.txt",    "w");
    int ch;
    while ((ch = fgetc(src)) != EOF)
        fputc(ch, dest);
    fclose(src);  fclose(dest);
    printf("File copied!\n");
    return 0;
}
Finding Average of Numbers in Sequential File
A classic sequential file example: write numbers to a file, then read them back and compute the average.
CStep 1: Write Numbers to File
#include <stdio.h>

int main() {
    FILE *fp = fopen("numbers.txt", "w");
    int nums[] = {45, 80, 65, 90, 72, 55, 88};
    for (int i = 0; i < 7; i++)
        fprintf(fp, "%d\n", nums[i]);
    fclose(fp);
    printf("Written 7 numbers to numbers.txt\n");
    return 0;
}
CStep 2: Read and Compute Average
#include <stdio.h>

int main() {
    FILE  *fp = fopen("numbers.txt", "r");
    int    num, count = 0;
    float  sum   = 0;

    if (fp == NULL) { printf("File not found!\n"); return 1; }

    printf("Numbers read: ");
    while (fscanf(fp, "%d", &num) == 1) {
        printf("%d ", num);
        sum += num;
        count++;
    }
    fclose(fp);

    if (count == 0) { printf("No numbers found!\n"); return 1; }

    printf("\nCount   : %d\n", count);
    printf("Sum     : %.0f\n", sum);
    printf("Average : %.2f\n", sum / count);
    return 0;
}
Output
Numbers read: 45 80 65 90 72 55 88
Count : 7
Sum : 495
Average : 70.71

Random Access File

fseek · ftell · rewind · fread · fwrite · binary files
What is Random Access?
Random access allows jumping directly to any position in a file without reading through preceding records. It is used with binary files where every record has a fixed size, so the position of any record can be calculated as:

position = record_number × sizeof(record)

Three key functions for positioning:
FunctionPurpose
fseek(fp, offset, whence)Move file position pointer to given offset
ftell(fp)Return current file position (in bytes)
rewind(fp)Move position to beginning of file
💡 fseek whence values: SEEK_SET (from beginning), SEEK_CUR (from current), SEEK_END (from end)
CExample 1 – Write & Read Binary File (fwrite / fread)
#include <stdio.h>

struct Employee {
    int   id;
    char  name[20];
    float salary;
};

int main() {
    struct Employee emp[3] = {
        {1, "Alice", 50000},
        {2, "Bob",   45000},
        {3, "Carol", 60000}
    };

    /* Write binary */
    FILE *fp = fopen("emp.dat", "wb");
    fwrite(emp, sizeof(struct Employee), 3, fp);
    fclose(fp);

    /* Read record #2 directly (index 1) */
    struct Employee e;
    fp = fopen("emp.dat", "rb");
    fseek(fp, 1 * sizeof(struct Employee), SEEK_SET);
    fread(&e, sizeof(struct Employee), 1, fp);
    printf("Record 2: ID=%d Name=%s Salary=%.0f\n",
           e.id, e.name, e.salary);
    fclose(fp);
    return 0;
}
Output
Record 2: ID=2 Name=Bob Salary=45000
CExample 2 – Update a Specific Record
#include <stdio.h>
#include <string.h>

struct Employee { int id; char name[20]; float salary; };

int main() {
    FILE *fp = fopen("emp.dat", "rb+");   // open for read+write
    struct Employee e;
    int recNo = 2;    // update 3rd record (index 2)

    /* Seek to record position */
    fseek(fp, recNo * sizeof(struct Employee), SEEK_SET);
    fread(&e, sizeof(e), 1, fp);
    printf("Before: %s - %.0f\n", e.name, e.salary);

    e.salary = 70000;    // update salary
    fseek(fp, recNo * sizeof(struct Employee), SEEK_SET);
    fwrite(&e, sizeof(e), 1, fp);
    printf("After : %s - %.0f\n", e.name, e.salary);
    fclose(fp);
    return 0;
}
Output
Before: Carol - 60000
After : Carol - 70000
CExample 3 – ftell and rewind
#include <stdio.h>

int main() {
    FILE *fp = fopen("numbers.txt", "r");
    printf("Position at start : %ld\n", ftell(fp));

    fseek(fp, 0, SEEK_END);
    printf("File size (bytes) : %ld\n", ftell(fp));

    rewind(fp);
    printf("After rewind      : %ld\n", ftell(fp));
    fclose(fp);
    return 0;
}
Output
Position at start : 0
File size (bytes) : 21
After rewind : 0

Command Line Arguments

argc · argv · main() parameters · shell input
What are Command Line Arguments?
Command line arguments allow passing values to a C program when it is run from the terminal/shell. These arguments are received by the main() function through two special parameters:
📌 argc (argument count) — number of arguments (including program name, so always ≥ 1)
📌 argv (argument vector) — array of strings (char*), where argv[0] is the program name
CSyntax
int main(int argc, char *argv[]) {
    // argc = number of arguments

CExample 1 – Print All Arguments
#include <stdio.h>

int main(int argc, char *argv[]) {
    printf("Number of arguments: %d\n", argc);
    for (int i = 0; i < argc; i++)
        printf("argv[%d] = %s\n", i, argv[i]);
    return 0;
}

/* Run: ./program Hello World 42 */
Output (run as: ./program Hello World 42)
Number of arguments: 4
argv[0] = ./program
argv[1] = Hello
argv[2] = World
argv[3] = 42
CExample 2 – Calculator using Command Line Args
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]) {
    if (argc != 4) {
        printf("Usage: %s num1 op num2\n", argv[0]);
        return 1;
    }
    float a = atof(argv[1]);
    char  op = argv[2][0];
    float b = atof(argv[3]);

    switch (op) {
        case '+': printf("%.2f\n", a + b); break;
        case '-': printf("%.2f\n", a - b); break;
        case '*': printf("%.2f\n", a * b); break;
        case '/':
            if (b == 0) printf("Division by zero!\n");
            else       printf("%.2f\n", a / b);
            break;
        default: printf("Unknown operator: %c\n", op);
    }
    return 0;
}
Usage Examples
$ ./calc 10 + 5 → 15.00
$ ./calc 20 - 8 → 12.00
$ ./calc 6 * 7 → 42.00
$ ./calc 15 / 4 → 3.75
$ ./calc 5 / 0 → Division by zero!
CExample 3 – File Copy using Command Line Args
#include <stdio.h>

int main(int argc, char *argv[]) {
    if (argc != 3) {
        printf("Usage: %s source.txt dest.txt\n", argv[0]);
        return 1;
    }
    FILE *src  = fopen(argv[1], "r");
    FILE *dest = fopen(argv[2], "w");
    if (!src || !dest) {
        printf("Error opening files!\n");
        return 1;
    }
    int ch;
    while ((ch = fgetc(src)) != EOF)
        fputc(ch, dest);
    printf("Copied '%s' to '%s'\n", argv[1], argv[2]);
    fclose(src); fclose(dest);
    return 0;
}

/* Run: ./copy notes.txt backup.txt */
Output
Copied 'notes.txt' to 'backup.txt'
CExample 4 – Sum of Numbers passed as Arguments
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char *argv[]) {
    float sum = 0;
    for (int i = 1; i < argc; i++)
        sum += atof(argv[i]);
    printf("Sum = %.2f\n", sum);
    return 0;
}

/* Run: ./sum 10 20 30 40 */
Output (run as: ./sum 10 20 30 40)
Sum = 100.00
